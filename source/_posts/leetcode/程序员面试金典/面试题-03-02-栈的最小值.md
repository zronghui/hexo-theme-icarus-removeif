class MinStack:

    def __init__(self):
        self.l = []
        self.minl = []

    def push(self, x: int) -> None:
        self.l.append(x)
        if not self.minl or x<=self.minl[-1]:
            self.minl.append(x) 

    def pop(self) -> None:
        if self.minl[-1]==self.l.pop():
            self.minl.pop()

    def top(self) -> int:
        return self.l[-1]

    def getMin(self) -> int:
        return self.minl[-1]



# Your MinStack object will be instantiated and called as such:
# obj = MinStack()
# obj.push(x)
# obj.pop()
# param_3 = obj.top()
# param_4 = obj.getMin()
